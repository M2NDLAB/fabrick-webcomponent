fbk-card · snippet 
<!-- 1. Includi lo script Fabrick nel <head> -->
<script src="https://sandbox-web.axerve.com/
  orchestra/checkout/assets/elements/
  fbk-elements.js"></script>

<!-- 2. Inserisci il custom element nella pagina -->
<fbk-card
  id="cardElement"
  style="width:100%"
  payment-id="{{ paymentID }}"
  payment-token="{{ paymentToken }}"
  layout="horizontal"
  optional-fields="true">
</fbk-card>

// 3. Configura shop e callback PRIMA di settare gli attributi
window['fbk-elements'] = {
  shop: "GESPAY12345",  // ← tuo shopLogin
  callBackObj: (event) => myCallbackFunction(event)
};

// 4. Funzione che riceve la risposta Fabrick al termine pagamento
// response.payload.flowStatus → COMPLETED | FAILED | EXPIRED
// response.payload.redirect   → oggetto con url e method
function myCallbackFunction(response) {
  const status = response.payload.flowStatus;
  if (status === "COMPLETED") {
    // redirect al tuo ordine/grazie
    window.location.href =
      response.payload.redirect.url;
  }
}

// 5. Imposta gli attributi dinamicamente (es. da backend)
const el = document.getElementById("cardElement");
el.setAttribute("payment-id",    paymentID);
el.setAttribute("payment-token", paymentToken);

// Configurazione globale condivisa (TUA)
// Va eseguita PRIMA che i componenti siano montati
window['fbk-elements'] = {
  shop: "GESPAY12345",
  callBackObj: (e) => myCallbackFunction(e)
};

// Risposta Fabrick – struttura completa:
// {
//   error: { code, description },
//   payload: {
//     flowStatus: "COMPLETED" | "FAILED" |
//                 "AUTHENTICATION_PENDING" | "EXPIRED",
//     redirect: { method, url, data? }
//   }
// }
// Dopo COMPLETED/FAILED → chiama payment/detail
function myCallbackFunction(response) {
  const { flowStatus, redirect }
    = response.payload;

  if (flowStatus === "COMPLETED") {
    // redirect verso la pagina di conferma
    window.location.href = redirect.url;

  } else if (flowStatus === "FAILED") {
    // mostra messaggio d'errore all'utente
    showError(response.error.description);

  } else if (flowStatus === "AUTHENTICATION_PENDING") {
    // 3DS: esegui il redirect per completare l'auth
    if (redirect.method === "POST") {
      submitPostForm(redirect.url, redirect.data);
    } else {
      window.location.href = redirect.url;
    }
  }
}
